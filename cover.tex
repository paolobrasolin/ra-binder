\newcommand\drawbaselinegrid{%
  \begin{tikzpicture}[overlay,remember picture,dotted,ultra thick]
    \coordinate (size ref) at (-\bL,-\bL);
    \coordinate (trim ref) at (   0,   0);
    \coordinate (safe ref) at ( \iL, \iL);
    \draw[black]   (size ref) rectangle ++(\wL+2\bL,\hL+2\bL);
    \draw[red]     (trim ref) rectangle ++(\wL,\hL);
    \draw[magenta] (safe ref) rectangle ++(\wL-2\iL,\hL-2\iL);
    \begin{scope}[even odd rule, gray]
      \fill (trim ref) rectangle ++(\wL,\hL)
            (size ref) rectangle ++(\wL+2\bL,\hL+2\bL);
    \end{scope}
    \draw [blue] (trim ref)++(\wL/2-\gL/2,0) rectangle ++(\gL,\hL);
  \end{tikzpicture}}
\makeatother
